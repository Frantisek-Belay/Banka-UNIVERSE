1    package org.example;        /** Toto je deklarace balíčku, ve kterém je třída `Banka5` uložena. Balíček slouží k organizaci tříd v rámci projektu. */

3    import java.util.Scanner;   /** Importuje třídu `Scanner` z balíčku `java.util`, která umožňuje čtení vstupu od uživatele. */
4    import java.util.InputMismatchException;    /** Importuje třídu `InputMismatchException` z balíčku `java.util`,
5                                                která reprezentuje výjimku při chybném vstupu od uživatele. */

7    public class Banka5 {   /** Začátek definice třídy `Banka5`. Třída obsahuje veškerý kód pro tuto bankovní aplikaci. */
8        private static BankovyUcet[] ucty = new BankovyUcet[5]; /** Deklarace a inicializace pole `ucty`, které bude uchovávat bankovní účty.
9                                                                V tomto případě je inicializováno pole s kapacitou 5 účtů. */
10   
11       public static void main(String[] args) {        /**  Začátek definice hlavní metody `main`. Tato metoda se spouští při spuštění programu */
12           Scanner scanner = new Scanner(System.in);   /**  Vytvoření instance třídy `Scanner` pro čtení vstupu od uživatele prostřednictvím standardního vstupu (`System.in`). */
13   
14           System.out.println("**********************************");
15           System.out.println("*    Vítejte v bance UNIVERSE    *");
16           System.out.println("**********************************");
17   
18           for (int i = 0; i < ucty.length; i++) {     /** Zde začíná cyklus pro vytvoření bankovních účtů.
19                                                       Procházíme pole `ucty` a pro každý účet se ptáme na jméno a vytváříme instanci třídy `BankovyUcet`,
20                                                       kterou ukládáme do pole. */
21               System.out.print("Zadejte jméno " + (i + 1) + ". účtu: ");
22               String jmeno = scanner.next();
23               ucty[i] = new BankovyUcet(jmeno);
24           }
25   
26           while (true) {  /** Toto je nekonečný cyklus, který umožňuje uživatelům pracovat s účty, dokud nezvolí možnost ukončení programu. */
27               System.out.println("**********************************");
28               System.out.println("*          HLAVNÍ MENU:          *");
29               System.out.println("**********************************");
30               System.out.println("*       1 - Výběr účtu           *");
31               System.out.println("*       2 - Konec programu       *");
32               System.out.println("**********************************");
33               System.out.print("Zadejte číslo vašeho požadavku: ");
34   
35               try {   /** Začátek bloku `try`, který obsahuje kód pro čtení volby od uživatele a zpracování chyb vstupu. */
36                   int volba = scanner.nextInt();  /** Čtení volby od uživatele jako celého čísla. */
37                   if (volba == 1) {   /** Větve `if` a `else if` reagují na volbu uživatele a volají příslušné metody pro práci s účty. */
38                       int cisloUctu = vyberCisloUctu(ucty, scanner);
39                       if (cisloUctu != -1) {
40                           pracujSUctem(ucty[cisloUctu], scanner);
41                       }
42                   } else if (volba == 2) {
43                       break;
44                   } else {
45                       System.out.println("Neplatná volba. Zkuste to znovu.");
46                   }
47               } catch (InputMismatchException e) {    /** Začátek bloku `catch`, který zpracovává výjimku, která nastane, pokud uživatel zadá neplatný vstup (např. nečíselný znak). */
48                   System.out.println("Chyba: Musíte zadat číslo z menu.");    /** Výpis chybové zprávy, pokud uživatel zadá neplatný vstup. */
49                   scanner.next(); /** Vyčistí zbytek vstupu od neplatného vstupu od uživatele.*/
50               }   /** Konec `catch` bloku a návrat na začátek hlavního nekonečného cyklu. */
51           }
52       }

54       public static int vyberCisloUctu(BankovyUcet[] ucty, Scanner scanner) { /** Začátek definice metody `vyberCisloUctu`, která umožňuje uživateli vybrat účet pro práci. */
55           System.out.println("**********************************");
56           System.out.println("*         Seznam účtů:           *");
57           System.out.println("**********************************");
58           for (int i = 0; i < ucty.length; i++) {
59               System.out.println((i + 1) + " - " + ucty[i].getJmeno());
60           }
61           System.out.println("**********************************");
62           System.out.println("Zadejte pořadové číslo účtu:");
63           System.out.println("**********************************");
64           int cisloUctu = scanner.nextInt();  /** Čtení čísla účtu od uživatele. */
65   
66           if (cisloUctu >= 1 && cisloUctu <= ucty.length) {   /** Větve `if` a `else` kontrolují platnost vstupu od uživatele a vracejí index vybraného účtu. */
67               System.out.println("Nyní pracujete s účtem č. " + cisloUctu + ". " + ucty[cisloUctu - 1].getJmeno());
68               return cisloUctu - 1;
69           } else {
70               System.out.println("Neplatné číslo účtu.");
71               return -1;
72           }
73       }

75       public static void pracujSUctem(BankovyUcet ucet, Scanner scanner) {    /** Začátek definice metody `pracujSUctem`, která umožňuje uživateli provádět operace s bankovním účtem. */
76           while (true) {
77               System.out.println("**********************************");
78               System.out.println("*        Podmenu účtu:           *");
79               System.out.println("**********************************");
80               System.out.println("* 1 - Vložit peníze              *");
81               System.out.println("* 2 - Vybrat peníze              *");
82               System.out.println("* 3 - Převod peněz na jiný účet  *");
83               System.out.println("* 4 - Výpis transakcí            *");
84               System.out.println("* 5 - Zobrazit zůstatek          *");
85               System.out.println("* 6 - Konec podmenu              *");
86               System.out.println("**********************************");
87   
88               System.out.print("Zadejte číslo vašeho požadavku: ");
89               int volba = scanner.nextInt();  /** Čtení volby od uživatele v rámci podmenu. */
90   
91               switch (volba) {    /** Blok `switch` reaguje na volbu uživatele a provádí příslušné operace s bankovním účtem. */
92                   case 1:     /** Větve `case` odpovídají různým operacím, jako je vkládání peněz, výběr peněz, převod peněz a další. */
93                       System.out.print("Zadejte částku k vložení: ");
94                       double castkaVklad = scanner.nextDouble();
95                       ucet.vlozPenize(castkaVklad);
96                       break;
97                   case 2:
98                       System.out.print("Zadejte částku k výběru: ");
99                       double castkaVyber = scanner.nextDouble();
100                      boolean uspesnyVyber = ucet.vyberPenize(castkaVyber);
101                      if (!uspesnyVyber) {
102                          System.out.println("Nedostatek prostředků na účtu.");
103                      }
104                      break;
105                  case 3:
106                      int cilovyUcetIndex = vyberCisloUctu(ucty, scanner);
107                      if (cilovyUcetIndex != -1) {
108                          BankovyUcet cilovyUcet = ucty[cilovyUcetIndex];
109                          System.out.print("Zadejte částku k převodu: ");
110                          double castkaPrevod = scanner.nextDouble();
111                          ucet.provedPřevod(cilovyUcet, castkaPrevod);
112                          System.out.println("Nyní pracujete s účtem č. " + (cilovyUcetIndex + 1) + ". " + cilovyUcet.getJmeno());
113                      }
114                      break;
115  
116                  case 4:
117                      ucet.vypisTransakce();
118                      break;
119                  case 5:
120                      System.out.println("Zůstatek na účtu: " + ucet.getZustatek());
121                      break;
122                  case 6:
123                      return;
124                  default:    /** `default` větev reaguje na neplatnou volbu uživatele. */
125                      System.out.println("Neplatná volba. Zkuste to znovu.");
126              } /** Návrat na začátek podmenu a možnost opakovat operace s účtem. */
127          }     /** Konec `while` smyčky pro podmenu. */
128      }
129  }             /** Konec definice třídy `Banka5`. */
